cfdisk {disk path}
    efi (300 MiB)
    root

mkfs.fat -F32 {efi partition path}
mkfs.ext4 {root partition path}

mount {root partition path} /mnt
mkdir /mnt/boot
mount {efi partition path} /mnt/boot

dd if=/mnt/dev/zero of=/mnt/swapfile bs=1G count=8 status=progress
chmod 0600 /mnt/swapfile
mkswap -U clear /mnt/swapfile
swapon /mnt/swapfile

pacstrap /mnt linux linux-firmware {intel or amd}-ucode base base-devel networkmanager nano

genfstab -U /mnt >> /mnt/etc/fstab

arch-chroot /mnt

bootctl install

nano /boot/loader/entries/arch.conf
    title Arch Linux
    linux /vmlinuz-linux
    initrd /{intel or amd}-ucode.img
    initrd /initramfs-linux.img
    options root={root partition path} rw

nano /etc/locale.gen
    en-US.UTF-8 UTF-8

locale-gen

echo LANG=en_US.UTF-8 > /etc/locale.conf

echo {host name} > /etc/hostname

ln -sf /usr/share/zoneinfo/America/New_York /etc/localtime

systemctl enable NetworkManager

timedatectl set-ntp true

passwd

exit

reboot

useradd -m {username} -G wheel

passwd {username}

    %wheel ALL=(ALL:ALL) ALL
